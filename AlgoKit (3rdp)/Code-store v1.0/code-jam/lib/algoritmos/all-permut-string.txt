// TODAS AS PERMUTACOES DE UMA STRING

#include <iostream>
#include <algorithm>
#include <stdio.h>
#include <stdlib.h>

using namespace std;

char s[50];
int v,len;
char n[50];

void st(int l,int ult) {
	if(l==v) printf("%s\n",n);
	else
	for(int i=ult+1;i!=len;i++) {
		if(i!=ult+1 && s[i]==s[i-1]) continue;
		n[l]=s[i]; st(l+1,i);
	}
}

int main() {

	while((cin >> s >> v)) {
		len=strlen(s);
		sort(&s[0],&s[len]);
		n[v] = '\0';
		st(0,-1);
	}

	return 0;

}
