#include <iostream>
#include <ext/hash_map> 
#include <ext/rope> 

// libstdc++/14648
void test01() 
{ 
  using namespace __gnu_cxx;

  typedef hash_map<char, crope, hash<char>, std::equal_to<char> > maptype; 
  maptype m; 
  m['l'] = "50";
  m['x'] = "10";
  std::cout << "m['x'] = " << m['x'] << std::endl; 
}  

int main()
{
  test01();
  return 0;
}
